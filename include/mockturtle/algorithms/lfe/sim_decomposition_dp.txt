/*! \brief Parameters for sim_decomposition algorithm */
struct sim_decomposition_dp_params
{
  bool verbose{false};
  size_t max_sup{2};
  bool is_informed{true};
  bool try_top_decomposition{true};
  bool try_bottom_decomposition{false};
  bool use_correlation{false};
};

namespace detail
{

  template<typename Ntk>
  class sim_decomposition_dp_impl
  {
    public:
      using TT = typename kitty::partial_truth_table;

    public:
      sim_decomposition_dp_impl( simulation_view<Ntk> & ntk, TT target, sim_decomposition_dp_params & ps )
      : ntk(ntk),
        target(target),
        ps(ps),
        F(target)
      {
      }
    
      size_t fanin_size( signal<klut_network> f )
      {
        size_t gate_count = 1;
        ntk.foreach_fanin( ntk.get_node( f ), [&]( auto const& s ) {
          gate_count += fanin_size( s );
        } );
        return gate_count;
      }

      signal<Ntk> idsd_step( std::vector<sim_pattern<Ntk>>& X, sim_pattern<Ntk> & Y )
      {
        
        if( X.size() == 0 )
          return ntk.get_constant( false );    
        if( X[0].pat.num_bits() == 0 )
          return ntk.get_constant( false );

        assert( ( X[0].pat.num_bits() == Y.pat.num_bits() ) );

        if( kitty::count_ones(Y.pat ) == 0 ) // contraddiction
          return ntk.get_constant( false );
        else if( kitty::count_ones(Y.pat) == Y.pat.num_bits() ) // tautology
          return ntk.get_constant( true );


        uint32_t bidx = 0;
        uint32_t bidx_min = 0;
        sim_pattern<Ntk> bpat;
        sim_pattern<Ntk> bpat_min;
        uint32_t idx_min = 0;
        double Inew = 0;
        double Imax = std::numeric_limits<double>::min();
        double Imin = std::numeric_limits<double>::max();
        std::vector<double> mi_vect;
        int64_t Nbits = int64_t(X[0].pat.num_bits());

        std::vector<uint32_t> idx_vect;

        std::vector<size_t> to_be_deleted;

        if( ps.is_informed )
        {
          for( size_t i = 0; i < X.size(); ++i )
          {
            if( ( kitty::count_ones(X[i].pat) == Nbits ) || ( kitty::count_ones(X[i].pat) == 0 ) )
            {
              to_be_deleted.push_back(i);
            }
            else
            { 
              if( ps.use_correlation )
              {
                Inew = std::abs(double(kitty::count_ones( X[i].pat&Y.pat ))-double(kitty::count_ones( X[i].pat )*kitty::count_ones( Y.pat ))/Nbits); 
              }
              else
              {
                Inew = kitty::mutual_information( std::vector{&(X[i].pat)}, &(Y.pat) );
              }

              idx_vect.push_back( i-to_be_deleted.size() );
              if( Inew >= Imax )
              {
                bidx = i;
                bpat = X[i];
                Imax = Inew;
              }
              if( Inew <= Imin )
              {
                bidx_min = i;
                bpat_min = X[i];
                Imin = Inew;
              }
            }
          }
          
          if( to_be_deleted.size() > 0 )
          {
            std::reverse(to_be_deleted.begin(), to_be_deleted.end());
            
            size_t count = 0;
            for( size_t i = 0; i < to_be_deleted.size() ; ++i )
            {
              if( to_be_deleted[i] <= bidx )
                count++;
              X.erase( X.begin()+to_be_deleted[i] ) ;
            }
            bidx -= count;
          }
        }

        if( X.size() == 0 )
          return ntk.get_constant( false );

        if( X.size() <= ps.max_sup )
        {
          std::vector<TT*> ipatterns;
          TT * opatterns = & Y.pat;
          std::vector<signal<Ntk>> children;
          for( size_t i = 0; i < X.size(); ++i )
          {
            children.push_back( X[i].sig );
            ipatterns.push_back( &(X[i].pat) );
          }

          chj_result chj_res = chatterjee_method( ipatterns, &(Y.pat) );
          signal<Ntk> fc = ntk.create_node( children, chj_res.dtt );
          if( ps.verbose )
          {
            std::cout << fc << " = ";
            for( size_t i = 0; i < children.size(); ++i )
              std::cout << children[i] << " ";
            kitty::print_binary(chj_res.dtt);
            std::cout << std::endl;
          }
          return fc;
        }

        std::pair< std::vector<sim_pattern<Ntk>>, sim_pattern<Ntk>> XY0 = compute_cofactor0( X, Y, bidx );
        std::pair< std::vector<sim_pattern<Ntk>>, sim_pattern<Ntk>> XY1 = compute_cofactor1( X, Y, bidx ); 

        if( ps.try_top_decomposition )
        {
          sim_top_decomposition res = is_top_decomposable( XY0, XY1 );
          if ( res != sim_top_decomposition::none )
          {
            switch ( res )
            {
              default:
                assert( false );
              case sim_top_decomposition::and_:
              {
                signal<klut_network> F1 = idsd_step( XY1.first, XY1.second );
                signal<klut_network> Fnew = ntk.create_and( bpat.sig, F1 );
                if( ps.verbose )
                  std::cout << Fnew << "=" << bpat.sig << "AND" << F1 << std::endl;
                return Fnew;
              }
              case sim_top_decomposition::or_:
              {
                signal<klut_network> F0 = idsd_step( XY0.first, XY0.second );
                signal<klut_network> Fnew = ntk.create_or( bpat.sig, F0 );
                if( ps.verbose )
                  std::cout << Fnew << "=" << bpat.sig << "OR" << F0 << std::endl;
                return Fnew;
              }
              case sim_top_decomposition::lt_:
              {
                signal<klut_network> F0 = idsd_step( XY0.first, XY0.second );
                signal<klut_network> Fnew = ntk.create_lt( bpat.sig, F0 );
                if( ps.verbose )
                  std::cout << Fnew << "=" << bpat.sig << "' AND " << F0 << std::endl;
                return Fnew;

              }
              case sim_top_decomposition::le_:
              {  
                signal<klut_network> F1 = idsd_step( XY1.first, XY1.second );
                signal<klut_network> Fnew = ntk.create_le( bpat.sig, F1 );
                if( ps.verbose )
                  std::cout << Fnew << "=" << bpat.sig << "' OR " << F1 << std::endl;
                return Fnew;
              }
              case sim_top_decomposition::xor_:
              {
                remove_column_and_invert( X, Y, bidx ); 
                signal<klut_network> Fxor = idsd_step( X, Y );
                signal<klut_network> Fnew = ntk.create_xor( bpat.sig, Fxor );
                if( ps.verbose )
                  std::cout << Fnew << "=" << bpat.sig << " XOR " << Fxor << std::endl;
                return Fnew;
              }
            }
          }
        }
        if( ps.try_bottom_decomposition )
        {
          bottom_res<Ntk> bres = is_bottom_decomposable( X, Y, Imax, mi_vect, idx_vect );
          if( bres.found )
          {
            std::cout << "+" ; 
            ntk.create_node( bres.children, bres.chj.dtt );
            return idsd_step( X, Y );
          }
        }

        std::pair< std::vector<sim_pattern<Ntk>>, sim_pattern<Ntk>> XY0_min = compute_cofactor0( X, Y, bidx_min );
        std::pair< std::vector<sim_pattern<Ntk>>, sim_pattern<Ntk>> XY1_min = compute_cofactor1( X, Y, bidx_min );
         
        
        if( kitty::entropy(std::vector{XY0_min.second.pat})==kitty::entropy(std::vector{XY1_min.second.pat}) )
        {
          signal<klut_network> F0_min = idsd_step( XY0_min.first, XY0_min.second );
          signal<klut_network> f0_min = ntk.create_and( ntk.create_not( bpat_min.sig ), F0_min );

          signal<klut_network> F1_min = idsd_step( XY1_min.first, XY1_min.second );
          signal<klut_network> f1_min = ntk.create_and( bpat_min.sig, F1_min );

          signal<klut_network> Fmin = ntk.create_or( f1_min, f0_min );
        
          return Fmin;
        }
        else
        {  
          signal<klut_network> F0 = idsd_step( XY0.first, XY0.second );
          signal<klut_network> f0 = ntk.create_and( ntk.create_not( bpat.sig ), F0 );

          signal<klut_network> F1 = idsd_step( XY1.first, XY1.second );
          signal<klut_network> f1 = ntk.create_and( bpat.sig, F1 );

          signal<klut_network> Fmax = ntk.create_or( f1, f0 );
          return Fmax;
        }
      }

      signal<Ntk> run()
      {
        std::vector<sim_pattern<Ntk>> X = ntk.sim_patterns;

        X.erase( X.begin() );
        X.erase( X.begin() );

        sim_pattern<Ntk> Y = F;

        return idsd_step( X, Y );
      }
    
    private:
      simulation_view<Ntk>& ntk;
      sim_decomposition_dp_params & ps;
      kitty::partial_truth_table target;
    public:
      sim_pattern<Ntk> F; 
      std::vector<double> Iactive;
}; /* class sim_decomposition_impl */
    
} /* namespace detail */




#pragma region sim_decomposition_dp
/*! \brief sim_decomposition_dp algorithm assembles a network bottom up.
 *
 * This method iteratively creates and adds new informative nodes. 
 * It takes an empty network, the simulation patterns of its input nodes and the target functon. 
 * Then, the rest of the network is created using the strategy described in the paper
 * "Learning complex boolean functions: Algorithms and applications." by Oliveira, A., & Sangiovanni-Vincentelli, A. (1993).
 * \param ntk the network wrapped in a simulation view.
 * \param examples the input patterns to be used for assembling the network
 * \param target target simulation pattern
 * \param ps parameters of the sim_decomposition_dp method
 *
 */
template<typename Ntk>
signal<Ntk> sim_decomposition_dp( simulation_view<Ntk>& ntk, std::vector<kitty::partial_truth_table> & examples, kitty::partial_truth_table const & target, sim_decomposition_dp_params & ps )
{
  ntk.initialize_network( examples );


  if( ps.verbose )
  {
    std::cout << "  ";
    for( size_t i = 0; i < ntk.sim_patterns[0].pat.num_bits(); ++i )
      std::cout << "-";
    std::cout << std::endl;
    for( auto x : ntk.sim_patterns )
    {
      std::cout << x.sig << " "; kitty::print_binary( x.pat ); std::cout << std::endl;
    }
    std::cout << "  ";
    for( size_t i = 0; i < ntk.sim_patterns[0].pat.num_bits(); ++i )
      std::cout << "-";
    std::cout << std::endl;
    std::cout << "y "; kitty::print_binary( target ); std::cout << std::endl;
    std::cout << "  ";
    for( size_t i = 0; i < ntk.sim_patterns[0].pat.num_bits(); ++i )
      std::cout << "-";
    std::cout << std::endl;
  }
  detail::sim_decomposition_dp_impl impl( ntk, target, ps );
  auto osignal = impl.run();
  return osignal;
}

template<typename Ntk>
std::vector<signal<Ntk>> sim_decomposition_dp( simulation_view<Ntk>& ntk, std::vector<kitty::partial_truth_table> & examples, std::vector<kitty::partial_truth_table> const & targets, sim_decomposition_dp_params & ps )
{
  std::vector<signal<Ntk>> osignals;
  ntk.initialize_network( examples );

  if( ps.verbose )
  {
    std::cout << "  ";
    for( size_t i = 0; i < ntk.sim_patterns[0].pat.num_bits(); ++i )
      std::cout << "-";
    std::cout << std::endl;
    for( auto x : ntk.sim_patterns )
    {
      std::cout << x.sig << " "; kitty::print_binary( x.pat ); std::cout << std::endl;
    }
    std::cout << "  ";
    for( size_t i = 0; i < ntk.sim_patterns[0].pat.num_bits(); ++i )
      std::cout << "-";
    std::cout << std::endl;
    for( auto y : targets )
    {
      std::cout << "  "; kitty::print_binary( y ); std::cout << std::endl;
    }
    std::cout << "  ";
    for( size_t i = 0; i < ntk.sim_patterns[0].pat.num_bits(); ++i )
      std::cout << "-";
    std::cout << std::endl;
  }

  for( uint32_t i = 0; i < targets.size(); ++i )
  {
    detail::sim_decomposition_dp_impl impl( ntk, targets[i], ps );
    osignals.push_back(impl.run());

    ntk.initialize_network( examples );
  }

  return osignals;
}
